/**
 * ENVIA O HISTÓRICO DE VIAGENS PRA O SUPABASE.
 *
 * Lê a aba Base, pega as viagens FINALIZADAS e manda pro Supabase (tabela
 * viagens_historico). Não duplica: usa o Rostering ID como chave (upsert).
 * Roda sozinho todo dia. Na 1ª execução já sobem os ~13 dias que a Base tem.
 *
 * COMO USAR (uma vez):
 *  1) Planilha → Extensões → Apps Script (pode ser no mesmo projeto dos outros).
 *  2) Cole este código.
 *  3) Em SUPA_SERVICE_KEY, cole a sua chave **service_role** (a "secret").
 *     Pega em: Supabase → Project Settings → API → Project API keys → service_role → Reveal.
 *     ⚠️ Essa chave fica SÓ aqui no script. Nunca no painel, nunca no chat.
 *  4) Confira NOME_ABA_BASE.
 *  5) Rode "enviarHistoricoSupabase" uma vez (autorize). Depois "agendarHistoricoSupabase".
 */

var SUPA_URL = 'https://dvbezzmfnvbpumnwrqhj.supabase.co';
var SUPA_SERVICE_KEY = 'COLE_AQUI_SUA_CHAVE_service_role';   // <-- secret, só aqui
var NOME_ABA_BASE = 'Base';

// Colunas da Base (1-indexado)
var C = {
  routeId: 1, rostering: 2, servico: 3, data: 4, trecho: 14, origem: 15,
  origemETD: 19, origemATD: 20, destino: 24, destinoATA: 26,
  estado: 39, substatus: 40, pacotes: 37, motivoCancel: 42, causaRaiz: 43
};

function enviarHistoricoSupabase() {
  var base = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(NOME_ABA_BASE);
  if (!base) { SpreadsheetApp.getUi().alert('Aba "' + NOME_ABA_BASE + '" não encontrada.'); return; }

  var dados = base.getDataRange().getValues();
  // Chave = protocolo + trecho (uma viagem multi-trecho tem VÁRIAS linhas, uma por
  // trecho — todas com o mesmo protocolo). Assim cada trecho é preservado e não
  // manda a mesma chave 2x no lote. Última ocorrência de cada chave vence.
  var mapa = {};
  for (var i = 1; i < dados.length; i++) {
    var r = dados[i];
    var estado = String(r[C.estado - 1] || '').trim().toLowerCase();
    if (estado !== 'finalizado' && estado !== 'cancelado') continue;   // finalizadas E canceladas
    var rost = String(r[C.rostering - 1] || '').trim();
    var routeId = String(r[C.routeId - 1] || '').trim();
    // chave: usa o Rostering ID (Protocolo); se vazio/"0", usa o Route ID (Travel ID).
    // As faixas de numeração não se sobrepõem, então não precisa de prefixo.
    var idBase = (rost && rost !== '0') ? rost : routeId;
    if (!idBase) continue;   // sem protocolo E sem route id → não dá pra chavear
    var trecho = String(r[C.trecho - 1] || '').trim();
    var chave = idBase + '|' + trecho;
    mapa[chave] = {
      chave: chave,
      rostering_id: rost,
      route_id: routeId,
      trecho: trecho,
      data: paraData(r[C.data - 1]),
      servico: txt(r[C.servico - 1]),
      origem: txt(r[C.origem - 1]),
      destino: txt(r[C.destino - 1]),
      saida_programada: txt(r[C.origemETD - 1]),
      saida_real: txt(r[C.origemATD - 1]),
      chegada: txt(r[C.destinoATA - 1]),
      estado: txt(r[C.estado - 1]),
      resultado: txt(r[C.substatus - 1]),
      pacotes: paraInt(r[C.pacotes - 1]),
      motivo_cancelamento: txt(r[C.motivoCancel - 1]),
      causa_raiz: txt(r[C.causaRaiz - 1])
    };
  }
  var linhas = Object.keys(mapa).map(function (k) { return mapa[k]; });

  var enviados = 0, erro = '';
  for (var j = 0; j < linhas.length; j += 500) {   // envia em lotes de 500
    var lote = linhas.slice(j, j + 500);
    var resp = UrlFetchApp.fetch(SUPA_URL + '/rest/v1/viagens_historico', {
      method: 'post',
      contentType: 'application/json',
      headers: {
        apikey: SUPA_SERVICE_KEY,
        Authorization: 'Bearer ' + SUPA_SERVICE_KEY,
        Prefer: 'resolution=merge-duplicates,return=minimal'   // upsert pelo rostering_id
      },
      payload: JSON.stringify(lote),
      muteHttpExceptions: true
    });
    var code = resp.getResponseCode();
    if (code >= 200 && code < 300) enviados += lote.length;
    else { erro = code + ': ' + resp.getContentText(); Logger.log(erro); break; }
  }
  var msg = erro ? ('ERRO — ' + erro) : ('Enviados ao Supabase: ' + enviados + ' de ' + linhas.length);
  SpreadsheetApp.getActiveSpreadsheet().toast(msg, erro ? 'Falhou' : 'OK', 10);
}

// Rode UMA vez: agenda o envio DUAS vezes por dia (~07h30 e ~19h30).
function agendarHistoricoSupabase() {
  ScriptApp.getProjectTriggers().forEach(function (t) {
    if (t.getHandlerFunction() === 'enviarHistoricoSupabase') ScriptApp.deleteTrigger(t);
  });
  ScriptApp.newTrigger('enviarHistoricoSupabase').timeBased().everyDays(1).atHour(7).nearMinute(30).create();
  ScriptApp.newTrigger('enviarHistoricoSupabase').timeBased().everyDays(1).atHour(19).nearMinute(30).create();
  SpreadsheetApp.getActiveSpreadsheet().toast('Agendado: envia ao Supabase ~07h30 e ~19h30.', 'OK', 6);
}

/* ---- helpers ---- */
function txt(v) {
  if (v == null || v === '') return null;
  if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm');
  return String(v);
}
function paraData(v) {
  if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  var m = String(v).match(/(\d{1,2})\/(\d{1,2})\/(\d{2,4})/);
  if (m) { var y = +m[3]; if (y < 100) y += 2000; return y + '-' + ('0' + m[2]).slice(-2) + '-' + ('0' + m[1]).slice(-2); }
  return null;
}
function paraInt(v) { var n = parseInt(String(v).replace(/[^\d-]/g, ''), 10); return isNaN(n) ? null : n; }
