/**
 * ENVIA O HISTÓRICO DE VIAGENS PRA O SUPABASE (ETD + ETA) — TUDO DA ABA BASE.
 *
 * A Base é a fonte central e tem os dois momentos da viagem:
 *  - ETA (chegada na origem): Origem ETA (Q) = deveria chegar · Origem ATA (R) = chegou.
 *  - ETD (viagem):            Origem ETD (S) = deveria sair   · Origem ATD (T) = saiu.
 *
 * Grava dois tipos de registro na tabela viagens_historico:
 *  - fase='ETD': FINALIZADAS e CANCELADAS (como antes). chave = protocolo|trecho.
 *  - fase='ETA': chegadas já realizadas (Origem ATA preenchida). chave = protocolo|trecho|ETA.
 * Não duplica: upsert pela chave (namespaces separados, nunca colidem).
 * Roda sozinho A CADA 30 MIN. Na 1ª execução já sobem os dias que a Base tem.
 *
 * COMO USAR (uma vez):
 *  1) Rode antes o SQL de `migracao-fase-eta-etd.md` (cria a coluna `fase`).
 *  2) Planilha → Extensões → Apps Script. Cole este código (substitui o anterior).
 *  3) Em SUPA_SERVICE_KEY, cole a sua chave **service_role** (a "secret").
 *     ⚠️ Essa chave fica SÓ aqui no script. Nunca no painel, nunca no chat.
 *  4) Confira NOME_ABA_BASE e as colunas em C (principalmente origemETA/origemATA).
 *  5) Rode "enviarHistoricoSupabase" uma vez (autorize). Depois "agendarHistoricoSupabase".
 */

var SUPA_URL = 'https://dvbezzmfnvbpumnwrqhj.supabase.co';
var SUPA_SERVICE_KEY = 'COLE_AQUI_SUA_CHAVE_service_role';   // <-- secret, só aqui
var NOME_ABA_BASE = 'Base';
// Aba de ocorrências (motivo lançado em sistema) — a ocorrência mora aqui, não na Base.
// Sem isso o histórico fica sem ocorrência (o calendário/Relatórios mostram "sem ocorrência").
// >>> Confira o NOME da aba e as colunas: A(1)=Protocolo, O(15)=Ocorrência. Ajuste se diferente.
var NOME_ABA_OCOR = 'Ocorrências';
var C_OCOR = { protocolo: 1, ocorrencia: 15 };

// Colunas da Base (1-indexado). Horários: Origem ETA=P(16) deveria chegar,
// Origem ATA=Q(17) chegou, Origem ETD=S(19) deveria sair, Origem ATD=T(20) saiu.
var C = {
  routeId: 1, rostering: 2, servico: 3, data: 4, trecho: 14, origem: 15,
  origemETA: 16, origemATA: 17, origemETD: 19, origemATD: 20,
  preCheckOrigem: 22, destino: 24, destinoETA: 25, destinoATA: 26,
  fimDescarga: 27, prazoDescarga: 28, preCheckDestino: 32,
  estado: 39, substatus: 40, pacotes: 37, motivoCancel: 42, causaRaiz: 43
};

function enviarHistoricoSupabase() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var base = ss.getSheetByName(NOME_ABA_BASE);
  if (!base) { ss.toast('Aba "' + NOME_ABA_BASE + '" não encontrada.', 'Falhou', 8); return; }
  var dados = base.getDataRange().getValues();
  var mapa = {};

  // Ocorrência por protocolo (aba Ocorrências). Vira a causa_raiz no histórico.
  var ocorPorProto = {};
  var abaOcor = ss.getSheetByName(NOME_ABA_OCOR);
  if (abaOcor) {
    var dOcor = abaOcor.getDataRange().getValues();
    for (var k = 1; k < dOcor.length; k++) {
      var pk = String(dOcor[k][C_OCOR.protocolo - 1] || '').trim();
      var oc = String(dOcor[k][C_OCOR.ocorrencia - 1] || '').trim();
      if (pk && oc) ocorPorProto[pk] = oc;
    }
  }

  for (var i = 1; i < dados.length; i++) {
    var r = dados[i];
    var rost = String(r[C.rostering - 1] || '').trim();
    var routeId = String(r[C.routeId - 1] || '').trim();
    var idBase = (rost && rost !== '0') ? rost : routeId;   // protocolo; senão route id
    if (!idBase) continue;
    var trecho = String(r[C.trecho - 1] || '').trim();

    // ---- ETD: finalizadas e canceladas ----
    var estado = String(r[C.estado - 1] || '').trim().toLowerCase();
    if (estado === 'finalizado' || estado === 'cancelado') {
      var kEtd = idBase + '|' + trecho;
      mapa[kEtd] = {
        chave: kEtd, fase: 'ETD',
        rostering_id: rost, route_id: routeId, trecho: trecho,
        data: paraData(r[C.data - 1]),
        servico: txt(r[C.servico - 1]),
        origem: txt(r[C.origem - 1]),
        destino: txt(r[C.destino - 1]),
        saida_programada: txt(r[C.origemETD - 1]),
        saida_real: txt(r[C.origemATD - 1]),
        chegada: txt(r[C.destinoATA - 1]),
        fim_descarga: txt(r[C.fimDescarga - 1]),      // AA = fim do descarregamento
        prazo_descarga: txt(r[C.prazoDescarga - 1]),  // AB = prazo pra descarregar
        estado: txt(r[C.estado - 1]),
        resultado: txt(r[C.substatus - 1]),
        pacotes: paraInt(r[C.pacotes - 1]),
        motivo_cancelamento: txt(r[C.motivoCancel - 1]),
        causa_raiz: ocorPorProto[idBase] || txt(r[C.causaRaiz - 1]),   // ocorrência (aba) tem prioridade; senão causa raiz da Base
        pre_check_origem: txt(r[C.preCheckOrigem - 1]),
        pre_check_destino: txt(r[C.preCheckDestino - 1]),
        destino_eta: txt(r[C.destinoETA - 1]),
        origem_eta: txt(r[C.origemETA - 1])
      };
    }

    // ---- ETA: chegou na origem (Origem ATA preenchida e Origem ETA de referência) ----
    var dATA = parseDT_(r[C.origemATA - 1]);   // R = chegou de verdade
    var dETA = parseDT_(r[C.origemETA - 1]);   // Q = deveria chegar
    if (dATA && dETA) {
      var dataISO = Utilities.formatDate(dATA, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      var kEta = idBase + '|' + trecho + '|ETA';
      mapa[kEta] = {
        chave: kEta, fase: 'ETA',
        rostering_id: rost, route_id: routeId, trecho: trecho,
        data: dataISO,
        servico: txt(r[C.servico - 1]),
        origem: txt(r[C.origem - 1]),
        destino: txt(r[C.destino - 1]),
        saida_programada: fmtDT_(dETA),   // referência: deveria chegar
        saida_real: '',
        chegada: fmtDT_(dATA),            // chegada real na origem
        estado: 'Finalizado',
        resultado: (dATA < dETA) ? 'No prazo' : 'Atrasado',
        pacotes: paraInt(r[C.pacotes - 1]),
        motivo_cancelamento: null,
        causa_raiz: null,
        pre_check_origem: txt(r[C.preCheckOrigem - 1]),
        pre_check_destino: txt(r[C.preCheckDestino - 1]),
        destino_eta: txt(r[C.destinoETA - 1]),
        origem_eta: txt(r[C.origemETA - 1])
      };
    }
  }

  var linhas = Object.keys(mapa).map(function (k) { return mapa[k]; });
  var enviados = 0, erro = '';
  for (var j = 0; j < linhas.length; j += 500) {   // envia em lotes de 500
    var lote = linhas.slice(j, j + 500);
    var resp = UrlFetchApp.fetch(SUPA_URL + '/rest/v1/viagens_historico', {
      method: 'post', contentType: 'application/json',
      headers: {
        apikey: SUPA_SERVICE_KEY, Authorization: 'Bearer ' + SUPA_SERVICE_KEY,
        Prefer: 'resolution=merge-duplicates,return=minimal'   // upsert pela chave
      },
      payload: JSON.stringify(lote), muteHttpExceptions: true
    });
    var code = resp.getResponseCode();
    if (code >= 200 && code < 300) enviados += lote.length;
    else { erro = code + ': ' + resp.getContentText(); Logger.log(erro); break; }
  }
  var msg = erro ? ('ERRO — ' + erro) : ('Enviados ao Supabase: ' + enviados + ' de ' + linhas.length);
  ss.toast(msg, erro ? 'Falhou' : 'OK', 10);
}

// Roda UMA vez: agenda o envio A CADA 30 MINUTOS (mantém o Relatório fresco).
function agendarHistoricoSupabase() {
  ScriptApp.getProjectTriggers().forEach(function (t) {
    if (t.getHandlerFunction() === 'enviarHistoricoSupabase') ScriptApp.deleteTrigger(t);
  });
  ScriptApp.newTrigger('enviarHistoricoSupabase').timeBased().everyMinutes(30).create();
  SpreadsheetApp.getActiveSpreadsheet().toast('Agendado: envia ao Supabase a cada 30 min.', 'OK', 6);
}

/* ---- helpers ---- */
function parseDT_(v) {
  if (v instanceof Date) return isNaN(v.getTime()) ? null : v;
  var s = String(v || '').trim();
  if (!s) return null;
  var m = s.match(/(\d{1,2})\/(\d{1,2})\/(\d{2,4})(?:[ T](\d{1,2}):(\d{2}))?/);
  if (m) {
    var y = +m[3]; if (y < 100) y += 2000;
    return new Date(y, (+m[2]) - 1, +m[1], m[4] ? +m[4] : 0, m[5] ? +m[5] : 0);
  }
  var d = new Date(s);
  return isNaN(d.getTime()) ? null : d;
}
function fmtDT_(d) {
  if (!(d instanceof Date) || isNaN(d.getTime())) return null;
  return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm');
}
function txt(v) {
  if (v == null || v === '') return null;
  if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm');
  return String(v);
}
function paraData(v) {
  if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  var m = String(v).match(/(\d{1,2})\/(\d{1,2})\/(\d{2,4})/);
  if (m) { var y = +m[3]; if (y < 100) y += 2000; return y + '-' + ('0' + m[2]).slice(-2) + '-' + ('0' + m[1]).slice(-2); }
  return null;
}
function paraInt(v) { var n = parseInt(String(v).replace(/[^\d-]/g, ''), 10); return isNaN(n) ? null : n; }
